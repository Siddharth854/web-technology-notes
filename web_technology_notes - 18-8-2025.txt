2). External / Internal CSS:
	Syntax:
		element{
		property: value;
		property: value;
		}
					<!DOCTYPE html>
				<html lang="en">
				<head>
					<meta charset="UTF-8">
					<meta name="viewport" content="width=device-width, initial-scale=1.0">
					<title>Internal css</title>
					<style>
						h1{
							color: red;
							border: 2px dashed green;
						}
						h2, h3{
							color: blue;
							border: 2px solid hotpink;
						}
						.bak{
							
							background-color: hotpink;
						}
					</style>
				</head>
				<body>
					<div class="bak">
						<h1>heading h1</h1>
						<h2>heading h2</h2>
					</div>
				</body>
				</html>
