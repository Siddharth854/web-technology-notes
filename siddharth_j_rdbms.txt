create table college(College_ID int(100), College_Name varchar(100), Location varchar(30));
create table Department(Dept_ID int(100), Dept_Name varchar(100), Instructor_ID int(100));
create table Instructor(Instructor_ID int(100), Instructor_Name varchar(100));
create table Course(Course_Id int(100), Course_Name varchar(100));
CREATE TABLE Student (
  Student_Id INT NOT NULL PRIMARY KEY,
  Name VARCHAR(100) NOT NULL,
  Enrollment_Year YEAR
);

CREATE TABLE Enrollment (
  Enrollment_ID INT NOT NULL PRIMARY KEY,
  Student_ID INT NOT NULL,
  Course_ID INT NOT NULL,
  Enrollment_Date DATE,
  Grade VARCHAR(10));
